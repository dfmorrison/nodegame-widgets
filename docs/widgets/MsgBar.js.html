  MsgBar.prototype.parse = function () {
    var msg = {};
    var that = this;
    var key = null;
    var value = null;
    this.table.forEach( function(e) {
      
        if (e.x === 0) {
          key = e.content;
          msg[key] = ''; 
        }
        else if (e.x === 1) {
          
          value = e.content.value;
          if (key === 'state' || key === 'data') {
            try {
              value = JSON.parse(e.content.value);
            }
            catch (ex) {
              value = e.content.value;
            }
          }
          
          msg[key] = value;
        }
    });
    var gameMsg = new GameMsg(msg);
    node.info(gameMsg, 'MsgBar sent: ');
    return gameMsg;
  };
